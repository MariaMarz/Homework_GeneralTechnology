2) Вывести название и стоимость товаров от 20 EUR. 

SELECT 
ProductName, Price -- выводим в таблицу Название продуктов и их цену
FROM [Products]
WHERE
Price>=20  -- пишем условие задачи
ORDER BY Price ASC -- это уже от себя, фильтрация цены по возрастанию

4) В упорядоченном по стоимости виде вывести название и стоимость товаров со скидкой в 9% от всех поставщиков, кроме поставщика 1.

SELECT ProductName, Price -- выводим стоимость и название товаров
FROM [Products]
WHERE 
NOT SupplierID=1 -- исключаем из списка постащика 1
AND
Price * .91 -- пишем условие для вывода цены со скидкой
ORDER BY Price ASC -- упорядочиваем список по стоимости

5) Вывести контактные имена клиентов, кроме тех, что из France и USA.

SELECT CustomerName
FROM [Customers]

NOT Country='France'
